18class AmazonReviewPageExtractor: 

19 def __init__(self, page_content: str): 

20 """ 

21 extracts all reviews for a html page content 

22 :param page_content: source of the web page 

23 """ 

24 self.data = page_content # raw page data 

25 self.reviews = list() # list of reviews 

26 self.extracts_reviews() # extract reviews 

27 

28 def extracts_reviews(self) -> None: 

29 """ 

30 extract the reviews from raw data to the reviews list 

31 Each review consists of a heading and the body. 

32 The text of the review is in a span of the class="cr-original-review-content" 

33 The reviews can be extracted with this property 

34 :return: None 

35 """ 

36 review_parts = self.data.count(review_part_start) # count review tokens 

37 if review_parts > 0: 

38 start_idx = self.data.find(review_part_start) # starting point 

39 end_idx = self.data.find(review_part_end, start_idx) # starting end point 

40 while start_idx != -1: # As long as there are still reviews 

41 # extract the header an find the body 

42 header = ( 

43 remove_html_code( 

44 self.data[start_idx + len(review_part_start) : end_idx] 

45 ) 

46 + ". " 

47 ) 

48 start_idx = self.data.find(review_part_start, end_idx) 

49 end_idx = self.data.find(review_part_end, start_idx) 

50 # extract the body 

51 content = remove_html_code( 

52 self.data[start_idx + len(review_part_start) : end_idx] 

53 ) 

54 start_idx = self.data.find(review_part_start, end_idx) 

55 end_idx = self.data.find(review_part_end, start_idx) 

56 # concat the header and the body, store into the review array 

57 self.reviews.append(header + content)

60def remove_html_code(text: str) -> str: 

61 """ 

62 used to remove al html special character 

63 :param text: review text 

64 :return: cleaned up review text 

65 """ 

66 result = re.sub(" +", " ", text.replace("<br>", " ").replace("\n", " ").strip()) 

67 return result

86def get_amazon_product_id(url: str) -> str: 

87 """ 

88 Extract the amazon product id from product or review web page 

89 :param url: url of product 

90 :return: product id 

91 """ 

92 start = url.find("/dp/") # search pattern for a product url 

93 count = 4 

94 if start == -1: 

95 start = url.find("/product-reviews/") # search pattern for a review page 

96 count = 17 

97 if start == -1: 

98 raise Exception("Failed to find the product id in the given url: " + url) 

99 end = url.find("/", start + count) 

100 result = url[start + count : end] 

101 return result

117def extract_amazon_review(thread_idx: int, product_id: str, data: list): 

118 """ 

119 thread function 

120 creates a driver and opens the url 

121 stores the reviews into shared memory data 

122 :param thread_idx: index of current thread 

123 :param product_id: product id from amazon, used to generate the review url 

124 :param data: shared memory array to save the web page source 

125 :return: None 

126 """ 

127 driver = webdriver.Chrome( 

128 options=get_chrome_options(), executable_path=binary_path 

129 ) # create driver engine 

130 driver.get( 

131 get_amazon_link(product_id, thread_idx + 1) 

132 ) # set browser to use this page 

133 time.sleep(6) # wait for page 

134 result = driver.page_source # extract page html source 

135 driver.quit() # close driver 

136 review_page = AmazonReviewPageExtractor(result) # initialize review extractor class 

137 data[thread_idx] = review_page.reviews # store extracted reviews in shared memory

140def extract_all_amazon_reviews( 

141 product_url: str, max_pages: int, max_threads: int 

142) -> list: 

143 """ 

144 main function to extract all amazon reviews for a given product url 

145 :param product_url: product url copied from web browser 

146 :param max_pages: page limit if there are more review pages 

147 :param max_threads: thread limit used for html extraction 

148 :return: array of reviews 

149 """ 

150 product_id = get_amazon_product_id(product_url) 

151 thread_data: List[str] = [""] * max_pages 

152 threads: List[threading.Thread] = list() 

153 last_review_page = max_pages 

154 for thread_iter in range(0, max_pages, max_threads): 

155 # create only a limited amount of threads 

156 for web_page in range(thread_iter, thread_iter + max_threads, 1): 

157 # create thread and start 

158 threads.append( 

159 threading.Thread( 

160 target=extract_amazon_review, 

161 args=(web_page, product_id, thread_data), 

162 ) 

163 ) 

164 threads[web_page].start() 

165 for web_page in range(thread_iter, thread_iter + max_threads, 1): 

166 # join threads and count the number of reviews 

167 threads[web_page].join() 

168 review_list = thread_data[web_page] 

169 number_of_reviews = len(review_list) 

170 if number_of_reviews == 0: 

171 last_review_page = web_page 

172 break 

173 if last_review_page != max_pages: 

174 break 

175 pages = thread_data[:last_review_page] 

176 # concat reviews 

177 result = list() 

178 for page in pages: 

179 result += page 

180 return result

183def check_product_urls(urls: List[str]) -> bool: 

184 """ 

185 Used to check the input amazon urls 

186 :param urls: array of amazon product urls 

187 :return: True if everything is fine and raises an exception if there was no product id 

188 """ 

189 if len(urls) == 0: 

190 return False 

191 else: 

192 for url_idx in range(len(urls)): 

193 url = urls[url_idx] 

194 product_id = get_amazon_product_id(url) 

195 if len(product_id) != 10: 

196 # After a random test, all product IDs have 10 characters 

197 raise Exception( 

198 "Failed to extract the product id for the " 

199 + str(url_idx + 1) 

200 + ". url: " 

201 + url 

202 ) 

203 return True

207 product_url: str, max_pages: int, max_threads: int, thread_idx: int, data: list

209 """

223def extract_reviews_for_products( 

224 products: List[str], max_pages: int, max_threads: int 

225) -> list: 

226 """ 

227 Main function 

228 Is used to extract all reviews from a list of amazon products in parallel. 

229 The extracted review pages can be limited via max_pages. 

230 In order to optimize the performance used, the number of threads can be set using max_threads. 

231 We recommend to use 10 threads per product --> for 3 products pass 30 threads as max_threads 

232 :param products: list of product urls 

233 :param max_pages: review page limit for each product 

234 :param max_threads: thread limit for the execution in parallel 

235 :return: List[List[Reviews]] for each product an array of reviews 

236 """ 

237 if check_product_urls(products): 

238 threads: List[threading.Thread] = list() 

239 data = [None] * len(products) 

240 threads_for_product = math.trunc(max_threads / len(products)) 

241 for product_num in range(len(products)): 

242 url = products[product_num] 

243 threads.append( 

244 threading.Thread( 

245 target=thread_extract_reviews_for_product, 

246 args=(url, max_pages, threads_for_product, product_num, data), 

247 ) 

248 ) 

249 threads[product_num].start() 

250 for product_num in range(len(products)): 

251 threads[product_num].join() 

252 return data 

253 else: 

254 return []
